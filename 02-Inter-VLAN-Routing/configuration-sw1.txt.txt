! SW1 Configuration
! ==================

enable
configure terminal
hostname SW1

! Create VLANs
vlan 10
 name HR
vlan 20
 name Finance
exit

! Access port - PC1 (Fa0/1) - VLAN 10
interface FastEthernet0/1
 switchport mode access
 switchport access vlan 10
 no shutdown
exit

! Access port - PC2 (Fa0/2) - VLAN 10
interface FastEthernet0/2
 switchport mode access
 switchport access vlan 10
 no shutdown
exit

! Trunk port to SW2 (Fa0/4)
interface FastEthernet0/4
 switchport mode trunk
 switchport trunk encapsulation dot1q
 switchport trunk allowed vlan 10,20
 no shutdown
exit

! Trunk port to Router R1 (Fa0/3)
interface FastEthernet0/3
 switchport mode trunk
 switchport trunk encapsulation dot1q
 switchport trunk allowed vlan 10,20
 no shutdown
exit

end
write memory
